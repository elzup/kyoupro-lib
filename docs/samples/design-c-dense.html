Dijkstra 法

単一始点・非負重み最短路 O((V+E) log V)
優先度キューで未確定のうち最小距離の頂点を順に確定。負辺不可（→ Bellman-Ford）。g[v] = (to, cost) の隣接リスト。

図解 — 確定順

(0)──5──▶(1)   start = 0
 │        ▲    init  dist=[0, ∞, ∞]
 2        1    pop0  1=5 2=2 →[0,5,2]
 ▼        │    pop2  1=2+1=3 →[0,3,2]
(2)───────┘    pop1  settled
0▶1:5 0▶2:2 2▶1:1     ans [0,3,2]
KOTLIN 
fun dijkstra(g: Array<MutableList<Pair<Int,Long>>>, s: Int): LongArray {
  val dist = LongArray(g.size){Long.MAX_VALUE/4}; dist[s]=0
  val pq = PriorityQueue<Pair<Long,Int>>(compareBy{it.first})
  pq.add(0L to s)
  while (pq.isNotEmpty()) {
    val (d,v) = pq.poll()
    if (d > dist[v]) continue           // 古い距離は捨てる
    for ((to,c) in g[v]) if (d+c < dist[to]) {
      dist[to]=d+c; pq.add(dist[to] to to) }
  }
  return dist
}

import heapq
def dijkstra(g, s):
    dist = [float('inf')]*len(g); dist[s]=0
    pq = [(0, s)]                        # (dist, vertex)
    while pq:
        d, v = heapq.heappop(pq)
        if d > dist[v]: continue         # 古い距離は捨てる
        for to, c in g[v]:
            if d+c < dist[to]:
                dist[to]=d+c; heapq.heappush(pq,(dist[to],to))
    return dist
JAVA 
static long[] dijkstra(List<long[]>[] g, int s) {  // g[v]={to,cost}
  long[] dist = new long[g.length];
  Arrays.fill(dist, Long.MAX_VALUE/4); dist[s]=0;
  var pq = new PriorityQueue<long[]>((a,b)->Long.compare(a[0],b[0]));
  pq.add(new long[]{0,s});
  while (!pq.isEmpty()) {
    long[] t=pq.poll(); long d=t[0]; int v=(int)t[1];
    if (d > dist[v]) continue;
    for (long[] e : g[v]) { int to=(int)e[0]; long c=e[1];
      if (d+c < dist[to]) { dist[to]=d+c; pq.add(new long[]{dist[to],to}); } }
  }
  return dist;
}

using ll=long long;                  // g[v]={to,cost}
vector<ll> dijkstra(vector<vector<pair<int,ll>>>& g, int s){
  vector<ll> dist(g.size(), 4e18); dist[s]=0;
  priority_queue<pair<ll,int>,vector<pair<ll,int>>,greater<>> pq;
  pq.push({0,s});
  while(!pq.empty()){
    auto[d,v]=pq.top(); pq.pop();
    if(d > dist[v]) continue;
    for(auto[to,c]:g[v]) if(d+c < dist[to]){
      dist[to]=d+c; pq.push({dist[to],to}); }
  }
  return dist;
}

罠

  • if(d>dist[v])continue 必須（古い距離を捨てる、無いと TLE）
  • 負辺は不可 → Bellman-Ford
  • INF は加算で溢れない値。C++ 4e18 / 他 LONG_MAX/4
  • 経路復元は prev[to]=v を終点から逆順
