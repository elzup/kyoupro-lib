Dijkstra 法

単一始点最短路 / single-source shortest path O((V+E) log V)

使う場面 辺の重みが 非負 のグラフで、ある頂点から全頂点への最短距離。負辺があるなら Bellman-Ford。

図解 — 優先度キューで距離が確定する順
(0)──5──▶(1)     start = 0
 │        ▲      ---------------------------------
 2        1      init   dist = [ 0,  ∞,  ∞ ]
 ▼        │      pop 0  relax 1=5, 2=2
(2)───────┘             dist = [ 0,  5,  2 ]
                 pop 2  relax 1 = 2+1 = 3  ( < 5 )
edges:                  dist = [ 0,  3,  2 ]
 0▶1 w5         pop 1  settled
 0▶2 w2         ---------------------------------
 2▶1 w1         answer  0→1 = 3,  0→2 = 2
Kotlin
import java.util.PriorityQueue

fun dijkstra(g: Array<MutableList<Pair<Int,Long>>>, s: Int): LongArray {
  val dist = LongArray(g.size) { Long.MAX_VALUE / 4 }
  dist[s] = 0
  val pq = PriorityQueue<Pair<Long,Int>>(compareBy { it.first })
  pq.add(0L to s)
  while (pq.isNotEmpty()) {
    val (d, v) = pq.poll()
    // 古い距離はスキップ
    if (d > dist[v]) continue
    for ((to, c) in g[v]) {
      if (d + c < dist[to]) {
        dist[to] = d + c
        pq.add(dist[to] to to)
      }
    }
  }
  return dist
}
Python
import heapq

def dijkstra(g, s):
    INF = float('inf')
    dist = [INF] * len(g)
    dist[s] = 0
    pq = [(0, s)]            # (dist, vertex)
    while pq:
        d, v = heapq.heappop(pq)
        if d > dist[v]:
            continue         # 古い距離はスキップ
        for to, c in g[v]:
            if d + c < dist[to]:
                dist[to] = d + c
                heapq.heappush(pq, (dist[to], to))
    return dist
Java
// g[v] = list of {to, cost}
static long[] dijkstra(List<long[]>[] g, int s) {
  long[] dist = new long[g.length];
  Arrays.fill(dist, Long.MAX_VALUE / 4);
  dist[s] = 0;
  PriorityQueue<long[]> pq =
    new PriorityQueue<>((a, b) -> Long.compare(a[0], b[0]));
  pq.add(new long[]{0, s});
  while (!pq.isEmpty()) {
    long[] t = pq.poll();
    long d = t[0]; int v = (int) t[1];
    if (d > dist[v]) continue;
    for (long[] e : g[v]) {        // e = {to, cost}
      int to = (int) e[0]; long c = e[1];
      if (d + c < dist[to]) {
        dist[to] = d + c;
        pq.add(new long[]{dist[to], to});
      }
    }
  }
  return dist;
}
C++
using ll = long long;
// g[v] = vector of {to, cost}
vector<ll> dijkstra(vector<vector<pair<int,ll>>>& g, int s) {
  vector<ll> dist(g.size(), 4e18);
  dist[s] = 0;
  priority_queue<pair<ll,int>,
    vector<pair<ll,int>>, greater<>> pq;
  pq.push({0, s});
  while (!pq.empty()) {
    auto [d, v] = pq.top(); pq.pop();
    if (d > dist[v]) continue;
    for (auto [to, c] : g[v]) {
      if (d + c < dist[to]) {
        dist[to] = d + c;
        pq.push({dist[to], to});
      }
    }
  }
  return dist;
}
罠 / pitfalls
  • 同じ頂点が複数回 push される → if (d > dist[v]) continue で古いものを捨てる（これが無いと TLE）
  • 負の辺があると正しく動かない → Bellman-Ford（負閉路検出も）
  • INF は加算で溢れない値に。C++ 4e18、他は LONG_MAX/4
  • 経路復元は prev[to]=v を更新し、終点から逆順にたどる
